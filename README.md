# Teste Itaú BBA front-end - 06/2021

# Setup do projeto
Angular CLI: 11.0.5 e Node: 14.15.3

# Clone o repositório
git clone https://github.com/brunoscosta17/teste-itau-bba

# Acesse o diretório
cd dir_do_projeto/teste-itau-bba

# Instale as dependencias
npm install

# Inicie o projeto
ng serve --open
